### Создан парсер студентов
Для добавления "боевых" тестовых данных сделал парсер студентов из нашей google таблицы успеваемости(парсит из локальной xlsx копии).
Добавляет в БД не только фио и группу студента, но ещё и генерирует 12 значный hex инвайт код для регистрации 

```python3

    def generate_short_code(self, student_id, group_id):
        hash_input = f"{student_id}:{group_id}:{self.salt}".encode()
        # Используем sha256 и берем первые 12 символов в верхнем регистре
        full_hash = hashlib.sha256(hash_input).hexdigest()
        return full_hash[:12].upper()

    def process_excel(self, file_path):
        xl = pd.ExcelFile(file_path)
        last_sheet = xl.sheet_names[-1]
        df = pd.read_excel(
            xl, 
            sheet_name=last_sheet, 
            usecols="B:C", 
            names=['fio', 'group_name']
        )
        
        conn = self.get_db_connection()
        cur = conn.cursor()

        try:
            for _, row in df.iterrows():
                fio = str(row['fio']).strip() if pd.notna(row['fio']) else None
                group_name = str(row['group_name']).strip() if pd.notna(row['group_name']) else None

                if not fio or not group_name or fio == 'nan':
                    continue
                cur.execute("""
                    INSERT INTO groups (group_name) 
                    VALUES (%s) 
                    ON CONFLICT (group_name) DO UPDATE SET group_name = EXCLUDED.group_name 
                    RETURNING group_id;
                """, (group_name,))
                group_id = cur.fetchone()[0]

                cur.execute("""
                    INSERT INTO students (student_name, group_id) 
                    VALUES (%s, %s) 
                    ON CONFLICT ON CONSTRAINT unique_student_group DO NOTHING 
                    RETURNING student_id;
                """, (fio, group_id))
                
                result = cur.fetchone()
                student_id = result[0] if result else None
                
                if not student_id:
                    cur.execute("SELECT student_id FROM students WHERE student_name = %s AND group_id = %s", (fio, group_id))
                    student_id = cur.fetchone()[0]

                invite_code = self.generate_short_code(student_id, group_id)

                cur.execute("UPDATE students SET invite_code = %s WHERE student_id = %s", (invite_code, student_id))
                print(f"[{last_sheet}] Добавлен: {fio} ({group_name}) -> {invite_code}")

            conn.commit()
            print(f"\nОбработка листа '{last_sheet}' завершена.")
```
### Новая система регистрации
Тк до регистрации пользователей все студенты парсятся в БД. Было решено валидировать регистрацию с помошью invite кодов, которые и создаются при парсинге таблицы. 
Был взят 12 значный hex code из student_id, group_id + соль для того, чтобы никто не смог угадать паттерн генерации заранее и подделать инвайт код.

### ___***Менеджер всех скриптов***___
Для того, чтобы парсеры было удобно запускать, был создан самомасштабирующийся менеджер скриптов
```python3

class ParserManager:
    def __init__(self):
        self.parsers_dir = Path(__file__).parent / "parsers"
        self.parsers = []

    def discover_parsers(self):
        if not self.parsers_dir.exists():
            logging.error(f"Папка {self.parsers_dir} не найдена!")
            return
        self.parsers = sorted([f for f in self.parsers_dir.glob("*.py") if f.name != "__init__.py"])

    def run_parser(self, parser_path):
        module_name = parser_path.stem
        logging.info(f"--- Запуск: {module_name} ---")
        try:
            spec = importlib.util.spec_from_file_location(module_name, parser_path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            if hasattr(module, 'main'):
                module.main()
        except Exception as e:
            logging.error(f"Ошибка в {module_name}: {e}")

    def start(self):
        self.discover_parsers()
        print("\n--- Список парсеров ---")
        for i, p in enumerate(self.parsers, 1):
            print(f"{i}. {p.name}")
        
        choice = input("\nВведите номера через запятую или диапазон (напр. 1,3-4): ")
            if choice == '0':
                break

            try:
                selected_indices = set()
                parts = [p.strip() for p in choice.split(',')]
                
                for part in parts:
                    if '-' in part:
                        start, end = map(int, part.split('-'))
                        selected_indices.update(range(start, end + 1))
                    else:
                        selected_indices.add(int(part))

                for idx in sorted(selected_indices):
                    if 1 <= idx <= len(self.parsers):
                        self.run_parser(self.parsers[idx-1])
                    else:
                        print(f"Номер {idx} вне диапазона.")
                        
            except ValueError:
                print("Ошибка: вводите только цифры, запятые или дефис.")
```

CREATE OR REPLACE VIEW view_next_lessons AS
WITH current_data AS (
    SELECT 
        g.group_id,
        g.group_name,
        s.schedule_id,
        sub.name AS subject_name,
        t.name AS teacher_name,
        s.lesson_type,
        s.room_info,
        lt.start_time,
        s.day_idx, 
        CASE WHEN extract(dow from now()) = 0 THEN 7 ELSE extract(dow from now()) END as today_idx
    FROM groups g
    JOIN schedules s ON g.group_id = s.group_id
    JOIN subjects sub ON s.subject_id = sub.subject_id
    JOIN teachers t ON s.teacher_id = t.teacher_id
    JOIN lesson_times lt ON s.lesson_num = lt.lesson_num
)
SELECT DISTINCT ON (group_id)
    group_id,
    group_name,
    subject_name,
    teacher_name,
    lesson_type,
    room_info,
    start_time
FROM current_data
WHERE today_idx = day_idx
  AND start_time > current_time
ORDER BY group_id, start_time ASC;
